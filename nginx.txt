server {
    listen 80;
    server_name all.tbznet.top;
    return 301 https://$host$request_uri;
}

server {
    listen 443 ssl http2;
    server_name all.tbznet.top;

    ssl_certificate     /var/lib/marzban/certs/all.tbznet.top/fullchain.pem;
    ssl_certificate_key /var/lib/marzban/certs/all.tbznet.top/privkey.pem;

    client_max_body_size 50M;
    proxy_read_timeout   300;
    proxy_connect_timeout 300;
    proxy_send_timeout   300;

    # Reverse proxy → Marzban panel (HTTPS backend)
    location / {
        proxy_pass https://127.0.0.1:4178;
        proxy_ssl_verify off;

        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto https;
    }

    # Emergency service → FastAPI
    location /emergency/ {
        proxy_pass http://127.0.0.1:5010/emergency/;
        
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto https;
    }
}
